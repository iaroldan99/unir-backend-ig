# VARIABLES REQUERIDAS
# =============================================

# Page Access Token de Meta
# Obtenerlo desde: https://developers.facebook.com/tools/explorer/
# Permisos necesarios: instagram_basic, instagram_manage_messages, pages_messaging
INSTAGRAM_ACCESS_TOKEN=your_page_access_token_here

# Token de verificación para webhooks (cualquier string secreto)
# Ejemplo: Genera uno con: uuidgen (Mac/Linux) o usa un string personalizado
INSTAGRAM_VERIFY_TOKEN=your_custom_verify_token_here

# Instagram Business Account ID
# Obtenerlo desde: https://graph.facebook.com/v18.0/{PAGE_ID}?fields=instagram_business_account
INSTAGRAM_PAGE_ID=your_instagram_business_account_id_here

# Puerto del servidor (opcional)
SERVER_PORT=8080
